func padToWidth(s string, w int) string {
    cw := lipgloss.Width(s)
    if cw >= w {
        runes := []rune(s)
        var out []rune
        width := 0
        for _, r := range runes {
            rw := lipgloss.Width(string(r))
            if width+rw > w {
                break

            out = append(out, r)
            width += rw

    }

}
