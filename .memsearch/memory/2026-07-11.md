```go
func loadEnv() {
    f, err := os.Open(".env")
    if err != nil { return }
    defer f.Close()
    scanner := bufio.NewScanner(f)
    for scanner.Scan() {
        line := strings.TrimSpace(scanner.Text())
        if line == "" || strings.HasPrefix(line, "#") { continue }
        if k, v, ok := strings.Cut(line, "="); ok {
            os.Setenv(strings.TrimSpace(k), strings.TrimSpace(v))
        }
    }
}
```
